import matplotlib.pyplot as plt
import pandas as pd
import numpy as np
import math

age = ["Di 20", "T 20 - di 25", "T 25 - di 30", "T 30 - di 35", "T 35 - di 50", "Trn 50"]
num_of_people_1971 = [118, 772, 288, 112, 112, 117]
num_of_people_1972 = [96, 727, 376, 117, 118, 165]

# Cc ch s (index) cho cc thanh (bar)
ind = np.arange(len(age))

#  ln ca thanh 
width = 0.35

# V
fig, ax = plt.subplots(figsize=(12, 6))

# V biu  thanh ngang
ax.barh(age, num_of_people_1972)
ax.set_xlabel('S ngi', fontweight = 'bold')
ax.set_ylabel(' tui', fontweight = 'bold')
ax.set_yticks(ind)
ax.set_yticklabels(age)

# Hm dn nhn cho cc thanh
def add_value_labels(ax, spacing=5):  
    # Vi mi thanh s t mt nhn
    for rect in ax.patches:
        # Xc nh v tr X, Y  t cc nhn
        x_value = rect.get_width()
        y_value = rect.get_y() + rect.get_height()/4

        # Khong cch gia thanh v nhn
        space = spacing

        # Dng x_value  lm nhn cho cc thanh
        label = "{:}".format(x_value)

        # To ch thch
        ax.annotate(
            label,                      # Dng label lm nhn
            (x_value, y_value),         # t label  bn tri cc thanh
            xytext=(space, 0),          # t lable cch thanh mt khong bng space
            textcoords="offset points") # Interpret `xytext` as offset in points

# Call the function above. All the magic happens there.
add_value_labels(ax)

ax.set_title(
    'S ngi nghin ma ty ti mt trung tm cai nghin vo nm 1972', 
    fontsize = 15,
    fontweight = 'bold',
    pad = 20,
    loc = 'center')
plt.show()
